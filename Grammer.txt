shift-expr  : expr ((LSHIFT|RSHIFT) expr)*

expr        : term ((PLUS|MINUS) term)*

term        : factor ((MUL|DIV|MOD) factor)*

factor      : (PLUS|MINUS) factor
            : bit-not

bit-not     : (TILDE) bit-not
            : power

power       : atom (POW factor)*

atom        : INT|FLOAT
            : LPAREN expr RPAREN
